<shell code design>
	0 ~ 39byte: dummy
	40 ~ 43byte: ret
	44 ~ 256: shell code

	shellcode 내에서 pipe(int pfd[2]) ->
	fork() -> dup2(pfd[0], STDOUT_FILENO) -> execl()
	-> read(pfd[0], buffer) -> socket() -> listen()
	-> accept() -> send(buffer) -> close(clnt_sock)
	-> close(serv_sock) -> close(pfd[0]) -> close(pfd[1])
</shell code design>
